# training.txt
# Python packages for the A->E training run: data_prep.py (SaT chunking) and
# train_scaled.py. Pins are the versions this pipeline was developed/tested on.
# (The repo's requirements.txt is an unrelated manim/jupyter freeze -- ignore it
# for training; use this file.)
#
# INSTALL torch FIRST, matched to your hardware, THEN `pip install -r training.txt`:
#   * AMD Strix Halo / ROCm  -> see STRIX_HALO.md (torch from the ROCm wheel index).
#   * Mac (MPS) / CPU / CUDA  -> pip install "torch==2.2.2"  (or the build for your box).
# The `torch>=2.2` line below is already satisfied by that install, so pip won't
# clobber your platform-specific build; a fresh box gets a default-index build.

torch>=2.2
numpy<2                # torch 2.2.x is not compatible with numpy 2.x
datasets==2.21.0       # streams the HuggingFace corpus mixture (data_prep.py)
transformers==4.57.6   # gpt2 tokenizer (AutoTokenizer) for the real chunker
wtpsplit==2.2.1        # SaT sentence segmenter -> the "SaT Capped" chunker in data_prep.py
                       #   downloads the sat-3l-sm model from the HF hub on first run
tqdm==4.68.4           # training progress bar (optional; --progress off to disable)
matplotlib==3.9.4      # plot_metrics.py / plot_comparison.py (post-run loss curves)
